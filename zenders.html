<table id="demo"></table>


<script>

function loadDoc() {

  var xhttp = new XMLHttpRequest();

  xhttp.onreadystatechange = function() {

    if (this.readyState == 4 && this.status == 200) {

      myFunction(this);

    }

  };

  xhttp.open("GET", "https://doberwt.github.io/zenders", true);

  xhttp.send();

}

function myFunction(xml) {

  var i;

  var xmlDoc = xml.responseXML;

  var table="<tr><th>Artist</th><th>Title</th><th>Price</th></tr>";

  var x = xmlDoc.getElementsByTagName("UPC");

  for (i = 0; i <x.length; i++) { 

    table += "<tr><td>" +

    x[i].getElementsByTagName("Zender")[0].childNodes[0].nodeValue +

    "</td><td>" +

    x[i].getElementsByTagName("TS")[0].childNodes[0].nodeValue +

    "</td><td>" +

    x[i].getElementsByTagName("TS_ID")[0].childNodes[0].nodeValue +

    "</td></tr>";

  }

  document.getElementById("demo").innerHTML = table;

}

</script>
